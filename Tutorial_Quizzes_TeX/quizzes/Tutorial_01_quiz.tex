\textbf{Time:} 10 minutes\\
\textbf{Format:} pen \& paper, no calculators

\subsection*{Q1 (probability + expectation)}
Let $X$ be a discrete random variable with $P(X=0)=1-p$ and $P(X=1)=p$.

\begin{enumerate}
\item Compute $E[X]$ and $\mathrm{Var}(X)$.
\item Show that for any constant $a$, $\mathrm{Var}(X+a)=\mathrm{Var}(X)$.
\end{enumerate}
